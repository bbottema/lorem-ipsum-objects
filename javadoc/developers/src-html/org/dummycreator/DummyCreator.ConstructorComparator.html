054        /**
055         * This is the main-method used to create a dummy of a certain class. It's called with the needed class. e.g. Integer i =
056         * createDummyOfClass(Integer.class)
057         * 
058         * @param <T>
059         * @param clazz
060         * @return
061         */
062        public static <T> T createDummyOfClass(final Class<T> clazz) {
063            Map<Class<?>, UsedInfo<?>> used_classes = new HashMap<Class<?>, UsedInfo<?>>();
064            if (Modifier.isAbstract(clazz.getModifiers()) || Modifier.isInterface(clazz.getModifiers())) {
065                if (ClassBinder.getBindingForClass(clazz) == null) {
066                    throw new IllegalArgumentException("Cant instantiate an abstract class or an interface. Please bind it into ClassBinder");
067                }
068            }
069            return createDummyOfClass(clazz, used_classes);
070        }

072        /**
073         * A helper for the main-method to avoid loops
074         * 
075         * @param <T>
076         * @param clazz
077         * @param used_classes
078         * @return
079         */
080        @SuppressWarnings("unchecked")
081        private static <T> T createDummyOfClass(final Class<T> clazz, final Map<Class<?>, UsedInfo<?>> used_classes) {
082            // List of Classes, we already used for population. By remembering, we can avoid looping
083            if (used_classes.get(clazz) == null || !used_classes.get(clazz).isPopulated()) {
084                // Check, if there is an objectbinding for this class
085                Object bind = ClassBinder.getBindingForClass(clazz);
086                if (bind != null && bind.getClass() == clazz) {
087                    return (T) bind;
088                }
089    
090                T ret = checkPrimitivesAndArray(clazz);
091                // If it was a special, we got ret != null
092                if (ret != null) {
093                    return ret;
094                }
095    
096                UsedInfo<T> usedInfo = new UsedInfo<T>();
097                usedInfo.setInstance(ret);
098                used_classes.put(clazz, usedInfo);
099    
100                // Has this class be bind?
101                ret = checkClassBinder(clazz, used_classes);
102                if (ret != null) {
103                    return ret;
104                }
105    
106                // Do we need to create a string?
107                if (clazz == String.class) {
108                    return (T) RandomCreator.getRandomString();
109                }
110    
111                // Is the class an enum?
112                if (clazz.isEnum()) {
113                    T[] enums = clazz.getEnumConstants();
114                    return enums[RandomCreator.getRandomInt(enums.length - 1)];
115                }
116                // Load the constructors
117                List<Constructor<?>> consts = ConstructorCache.getCachedConstructors(clazz);
118                if (consts == null) {
119                    consts = new ArrayList<Constructor<?>>();
120                    Constructor<?>[] _con = clazz.getConstructors();
121                    // Sort the constructors by their parameter-count
122                    java.util.Arrays.sort(_con, new ConstructorComparator());
123                    consts.addAll(Arrays.asList(_con));
124                    // Add to cache
125                    ConstructorCache.addConstructors(clazz, consts);
126                }
127                // Check if we have a prefered Constructor and try it
128                Constructor<T> preferedConstructor = (Constructor<T>) ConstructorCache.getPreferedConstructor(clazz);
129                if (preferedConstructor != null) {
130                    ret = tryConstructor(preferedConstructor, used_classes);
131                }
132                if (ret == null) {
133                    for (Constructor<?> co : consts) {
134                        ret = (T) tryConstructor(co, used_classes);
135                        if (ret != null) {
136                            ConstructorCache.setPreferedConstructor(clazz, co);
137                            // Worked
138                            break;
139                        }
140    
141                    }
142                }
143                if (ret == null) {
144                    throw new IllegalArgumentException("The given class couldn't be instantiated");
145                }
146    
147                usedInfo.setInstance(ret);
148                usedInfo.setPopulated(true);
149                populateObject(ret, used_classes);
150    
151                return ret;
152            } else {
153                return (T) used_classes.get(clazz).getInstance();
154            }
155        }

157        /**
158         * A method, which tries to instanciate an object with the given constructor
159         * 
160         * @param <T>
161         * @param c
162         * @param used_classes
163         * @return
164         */
165        private static <T> T tryConstructor(final Constructor<T> c, final Map<Class<?>, UsedInfo<?>> used_classes) {
166            @SuppressWarnings("unchecked")
167            Class<T>[] parameters = (Class<T>[]) c.getParameterTypes();
168            try {
169                if (parameters.length > 0) {
170                    final Object[] params = new Object[parameters.length];
171                    for (int i = 0; i < params.length; i++) {
172                        params[i] = createDummyOfClass(parameters[i], used_classes);
173                    }
174                    return c.newInstance(params);
175                } else {
176                    return c.newInstance();
177                }
178            } catch (InvocationTargetException ite) {
179                // ite.printStackTrace();
180            } catch (InstantiationException ie) {
181                // ie.printStackTrace();
182            } catch (IllegalAccessException iae) {
183                // iae.printStackTrace();
184            }
185            // If this didn't work, we return null
186            return null;
187        }

189        /**
190         * Populates given object with dummy value. The behavior can vary depending on the type of object, as follows:
191         * <ul>
192         * <li><strong>Collection</strong>: if the type is a subtype of {@link Collection}, a random number of items, 1 or 2, will be added</li>
193         * <li><strong>Map</strong>: if the type is a subtype of {@link Map}, a random number of key/value entries, 1 or 2, will be added</li>
194         * <li><strong>Other types</strong>: The <em>setter</em> methods will be retrieved from the object and will be invoked with a new dummy
195         * value.</li>
196         * </ul>
197         * 
198         * @param subject The object to populate with dummy values.
199         * @param knownInstances A list of known instances to keep track of already processed classes (to avoid infinite loop)
200         * @see #createDummyOfClass(Class, Map)
201         */
202        @SuppressWarnings({ "unchecked" })
203        private static <T> void populateObject(final T subject, final Map<Class<?>, UsedInfo<?>> knownInstances) {
204            final Class<?> clazz = subject.getClass();
205    
206            if (subject instanceof Collection) {
207                for (int i = 0; i < getRandomArrayLength(2); i++) {
208                    // detect generic declarations
209                    Type[] genericTypes = ((ParameterizedType) subject.getClass().getGenericSuperclass()).getActualTypeArguments();
210                    if (genericTypes.length > 0 && (genericTypes[0] instanceof Class)) {
211                        // uses generic type if available, Integer for '<T extends List<Integer>>'
212                        ((Collection<Object>) subject).add(createDummyOfClass((Class<?>) genericTypes[0], knownInstances));
213                    } else {
214                        // use default String value for raw type Collection
215                        ((Collection<Object>) subject).add(createDummyOfClass(String.class, knownInstances));
216                    }
217                }
218            } else if (subject instanceof Map) {
219                for (int i = 0; i < getRandomArrayLength(2); i++) {
220                    // detect generic declarations
221                    Type[] genericTypes = ((ParameterizedType) subject.getClass().getGenericSuperclass()).getActualTypeArguments();
222                    if (genericTypes.length > 0 && (genericTypes[0] instanceof Class)) {
223                        // uses generic type if available, String and Integer for '<T extends Map<String, Double>>'
224                        ((Map<Object, Object>) subject).put(createDummyOfClass((Class<?>) genericTypes[0], knownInstances), createDummyOfClass((Class<?>) genericTypes[1], knownInstances));
225                    } else {
226                        // use default String and String value for raw type Collection
227                        ((Map<Object, Object>) subject).put(createDummyOfClass(String.class, knownInstances), createDummyOfClass(String.class, knownInstances));
228                    }
229                }
230            } else {
231                List<Method> setter = MethodCache.getSetterForClass(clazz);
232                if (setter == null) {
233                    setter = new ArrayList<Method>();
234                    for (Method m : clazz.getMethods()) {
235                        if (isSetter(m)) {
236                            setter.add(m);
237                        }
238                    }
239                    MethodCache.addSetterForClass(clazz, setter);
240                }
241    
242                Object parameter = null;
243                for (Method m : setter) {
244                    // Load the parameter to pass to this method
245                    parameter = createDummyOfClass(m.getParameterTypes()[0], knownInstances);
246                    try {
247                        m.invoke(subject, parameter);
248                    } catch (Exception e) {
249                        e.printStackTrace();
250                    }
251                }
252            }
253        }

255        @SuppressWarnings("unchecked")
256        private static <T> T checkClassBinder(final Class<T> clazz, final Map<Class<?>, UsedInfo<?>> used_classes) {
257            Object bind = ClassBinder.getBindingForClass(clazz);
258            T ret = null;
259            if (bind != null) {
260                // Do we have a constructor binding?
261                if (bind instanceof Constructor) {
262                    ret = tryConstructor((Constructor<T>) bind, used_classes);
263                    return ret;
264                } else // Was this class bind to a method?
265                if (bind instanceof Method) {
266                    Method m = (Method) bind;
267                    Class<?>[] parameters = m.getParameterTypes();
268                    final Object[] params = new Object[parameters.length];
269                    for (int i = 0; i < params.length; i++) {
270                        params[i] = createDummyOfClass(parameters[i], used_classes);
271                    }
272                    try {
273                        return (T) m.invoke(null, params);
274                    } catch (InvocationTargetException ite) {
275                        // ite.printStackTrace();
276                    } catch (IllegalAccessException iae) {
277                        // iae.printStackTrace();
278                    }
279                } else // Was this an interface?
280                if (clazz.isInterface()) {
281                    // Load class to use
282                    Class<? extends T> c = (Class<? extends T>) bind;
283                    return createDummyOfClass(c, used_classes);
284                }
285            }
286            return ret;
287        }

289        /**
290         * This method checks, if the class, we want to create is a special one (primitives and String)
291         * 
292         * @param <T>
293         * @param clazz
294         * @return
295         */
296        @SuppressWarnings("unchecked")
297        private static <T> T checkPrimitivesAndArray(final Class<T> clazz) {
298            // Check if we have a primitive or string
299            if (clazz.isPrimitive()) {
300                return buildPrimitive(clazz);
301            }
302            // Do we have an array?
303            if (clazz.isArray()) {
304                int length = getRandomArrayLength(20);
305                Object parameter = Array.newInstance(clazz.getComponentType(), length);
306                for (int i = 0; i < length; i++) {
307                    Array.set(parameter, i, createDummyOfClass(clazz.getComponentType()));
308                }
309                return (T) parameter;
310            }
311            return null;
312        }

314        /**
315         * Method to check, if the given method is a Setter. A setter is defined by the name setXXXX and by having only one parameter
316         * 
317         * @param method
318         * @return
319         */
320        private static boolean isSetter(Method method) {
321            if (!method.getName().startsWith("set")) {
322                return false;
323            }
324            if (method.getParameterTypes().length != 1) {
325                return false;
326            }
327            return true;
328        }

330        /**
331         * Method to build a primitive
332         * 
333         * @param c
334         * @return
335         */
336        @SuppressWarnings("unchecked")
337        private static <T> T buildPrimitive(Class<T> c) {
338            if (c == (java.lang.Integer.TYPE)) {
339                return (T) (Integer) RandomCreator.getRandomInt();
340            } else if (c == (java.lang.Long.TYPE)) {
341                return (T) (Long) RandomCreator.getRandomLong();
342            } else if (c == (java.lang.Float.TYPE)) {
343                return (T) (Float) RandomCreator.getRandomFloat();
344            } else if (c == (java.lang.Boolean.TYPE)) {
345                return (T) (Boolean) RandomCreator.getRandomBoolean();
346            } else if (c == (java.lang.Character.TYPE)) {
347                return (T) (Character) RandomCreator.getRandomChar();
348            } else if (c == (java.lang.Byte.TYPE)) {
349                return (T) (Byte) RandomCreator.getRandomByte();
350            } else if (c == (java.lang.Short.TYPE)) {
351                return (T) (Short) RandomCreator.getRandomShort();
352            } else if (c == (java.lang.Double.TYPE)) {
353                return (T) (Double) RandomCreator.getRandomDouble();
354            }
355            return null;
356        }

358        /**
359         * Method to generate a random length for an array
360         * 
361         * @param max TODO
362         * @return
363         */
364        private static int getRandomArrayLength(int max) {
365            Random rand = new Random();
366            return rand.nextInt(max) + 1;
367        }

369        /**
370         * Comparator to sort constructors by their number of parameters
371         */
372        private static class ConstructorComparator implements Comparator<Constructor<?>> {
373    
374            public int compare(Constructor<?> o1, Constructor<?> o2) {
375                int num_o1 = o1.getParameterTypes().length;
376                int num_o2 = o2.getParameterTypes().length;
377                if (num_o1 < num_o2) {
378                    return -1;
379                }
380                if (num_o1 == num_o2) {
381                    return 0;
382                }
383                return 1;
384            }
385        }
