015    package org.dummycreator;
016    
017    import java.util.Random;
018    
019    import de.svenjacobs.loremipsum.LoremIpsum;
020    
021    /**
022     * Utility class that contains various methods to generate dummy data for primitive types.
023     * 
024     * @author Alexander Muthmann <amuthmann@dev-eth0.de> (original author)
025     * @author Benny Bottema <b.bottema@projectnibble.org> (further developed project)
026     */
027    final class RandomCreator {
028    
029        private static final Random r = new Random();
030        private static final LoremIpsum loremIpsum = new LoremIpsum();
031    
032        private RandomCreator() {
033            // utility class, can't instantiate
034        };
035    
036        public static String getRandomString() {
037            String gibberish = "";
038            for (int i = 0; i <= r.nextInt(3); i++) {
039                gibberish += (gibberish.length() != 0 ? " " : "") + loremIpsum.getWords(1, r.nextInt(50));
040            }
041            return gibberish.replaceAll("\\s", "_").replaceAll("[\\.\\,]", "");
042        }
043    
044        public static boolean getRandomBoolean() {
045            return r.nextBoolean();
046        }
047    
048        public static int getRandomInt() {
049            return r.nextInt();
050        }
051    
052        public static char getRandomChar() {
053            return (char) r.nextInt();
054        }
055    
056        public static byte getRandomByte() {
057            return (byte) r.nextInt();
058        }
059    
060        public static long getRandomLong() {
061            return r.nextLong();
062        }
063    
064        /**
065         * @return A float value between 00.00 and 100.00
066         */
067        public static float getRandomFloat() {
068            return r.nextInt(100) + ((float) Math.round(r.nextFloat() * 100) / 100);
069        }
070    
071        /**
072         * @return A double value between 00.00 and 100.00
073         */
074        public static double getRandomDouble() {
075            return r.nextInt(100) + ((double) Math.round(r.nextDouble() * 100) / 100);
076        }
077    
078        public static short getRandomShort() {
079            return (short) r.nextInt();
080        }
081    
082        public static int getRandomInt(int max) {
083            return r.nextInt(max);
084        }
085    }
