018    package org.dummycreator.binder;
019    
020    import java.lang.reflect.Constructor;
021    import java.lang.reflect.Method;
022    import java.lang.reflect.Modifier;
023    import java.util.ArrayList;
024    import java.util.HashMap;
025    import java.util.HashSet;
026    import java.util.List;
027    import java.util.Map;
028    import java.util.Set;
029    
030    /**
031     * 
032     * @author Alexander Muthmann <amuthmann@dev-eth0.de>
033     * @version 04/2010
034     */
035    public class ClassBinder {
036    
037        private static final HashMap<Class<?>, Object> bindings = new HashMap<Class<?>, Object>();
038    
039        private ClassBinder() {
040        }
041    
042        public static <T> void bind(final Class<T> _class, final Constructor<T> _constructor) {
043            bindings.put(_class, _constructor);
044        }
045    
046        public static <T> void bind(final Class<T> _interface, final Class<? extends T> _implementation) {
047            bindings.put(_interface, _implementation);
048        }
049    
050        public static <T> void bind(final Class<T> clazz, final Method method) {
051            if (Modifier.isStatic(method.getModifiers()) && method.getReturnType().equals(clazz)) {
052                bindings.put(clazz, method);
053            } else {
054                throw new IllegalArgumentException("The method has to be static and return an object of the given class!");
055            }
056        }
057    
058        public static <T> void bind(final Class<T> clazz, final Object object) {
059            // Check if the object is a subclass of clazz
060            if (clazz.isAssignableFrom(object.getClass())) {
061                bindings.put(clazz, object);
062            } else {
063                throw new IllegalArgumentException("The object has to have a subclass of clazz");
064            }
065        }
066    
067        /**
068         * This methods allows you to release a binding for a certain class
069         * 
070         * @param <T>
071         * @param _class
072         */
073        public static <T> void releaseBinding(final Class<T> _class) {
074            bindings.remove(_class);
075        }
076    
077        /**
078         * This method returns a binding made for the given class. This binding might be of one of the following type: Constructor
079         * Implementation of a Interface Method Object
080         * 
081         * @param _class
082         * @return
083         */
084        public static Object getBindingForClass(final Class<?> _class) {
085            return bindings.get(_class);
086        }
087    
088        /**
089         * You can call this method to build some default bindings for common classes. This includes List.class, Map.class, Set.class
090         */
091        public static void setDefaultBindings() {
092            ClassBinder.bind(List.class, ArrayList.class);
093            ClassBinder.bind(Map.class, HashMap.class);
094            ClassBinder.bind(Set.class, HashSet.class);
095            // TODO: Create-Methods for ArrayList, HashMap and HashSet would be great
096        }
097    }
