042    /**
043     * Tool to create populated dummy objects of a given class. This tool will recursively run through its setters and try to come up with newly
044     * populated objects for those fields.
045     * <p>
046     * To avoid recursive infinite loops, this tool keeps track of previously populated instances of a certain type and reuse that instead.
047     * <p>
048     * For numbers being generated a random number is being using, for strings a lorem ipsum generated is being used.
049     * 
050     * @see #create(Class)
051     * 
052     * @author Alexander Muthmann <amuthmann@dev-eth0.de>
053     * @author Benny Bottema <b.bottema@projectnibble.org>
054     */
055    public class DummyCreator {
056    
057        private final Logger logger = Logger.getLogger(getClass());
058    
059        /**
060         * A cache for previously found {@link Constructor} instances and preferred constructor for previous successfully invoked constructors.
061         */
062        private final ConstructorCache constructorCache;
063    
064        /**
065         * A cache for previously found {@link Method} instances and preferred constructor for previous successfully invoked constructors.
066         */
067        private final MethodCache methodCache;

096            this.methodCache = methodCache;
097            this.classBindings = classBindings;

100        /**
101         * Main method, creates a dummy object of a given type.
102         * <p>
103         * Provide your own {@link ClassBindings} in {@link #DummyCreator(ClassBindings)} to control how objects are created for specific types
104         * (such as the abstract List class). This is the main-method used to create a dummy of a certain class. It's called with the needed
105         * class. e.g. Integer i = createDummyOfClass(Integer.class)
106         * 
107         * @param <T> The type to be created and returned (returned type can be a sub type of <code>T</code>).
108         * @param clazz The type that should be created
109         * @return The instantiated and populated object (can be a sub type, depending how the {@link ClassBindings} are configured!).
110         * @throws IllegalArgumentException Thrown if an abstract type or interface was given for which no binding could be found in the
111         *             provided {@link ClassBindings}.
112         */
113        public <T> T create(final Class<T> clazz) {
114            Map<Class<?>, ClassUsageInfo<?>> used_classes = new HashMap<Class<?>, ClassUsageInfo<?>>();
115            if (Modifier.isAbstract(clazz.getModifiers()) || Modifier.isInterface(clazz.getModifiers())) {
116                if (classBindings.find(clazz) == null) {
117                    throw new IllegalArgumentException("Can't instantiate an abstract class or an interface. Please register it to ClassBindings");
118                }
119            }
120            return create(clazz, used_classes);
121        }

123        /**
124         * Will try to create a new object for the given type, while maintaining a track record of already created - and - populated objects to
125         * avoid recursive loops.
126         * 
127         * @param <T> The type to be created and returned (returned type can be a sub type of <code>T</code>).
128         * @param clazz The type that should be created
129         * @param knownInstances A list of previously created and populated objects for a specific type.
130         * @return The instantiated and populated object (can be a sub type, depending how the {@link ClassBindings} are configured).
131         */
132        @SuppressWarnings("unchecked")
133        private <T> T create(final Class<T> clazz, final Map<Class<?>, ClassUsageInfo<?>> knownInstances) {
134            // List of Classes, we already used for population. By remembering, we can avoid looping
135            if (knownInstances.get(clazz) == null || !knownInstances.get(clazz).isPopulated()) {
136                Object bind = classBindings.find(clazz);
137                if (bind != null && bind.getClass() == clazz) {
138                    return (T) bind;
139                }
140    
141                T ret = checkPrimitivesAndArray(clazz);
142                // If it was a special, we got ret != null
143                if (ret != null) {
144                    return ret;
145                }
146    
147                ClassUsageInfo<T> usedInfo = new ClassUsageInfo<T>();
148                usedInfo.setInstance(ret);
149                knownInstances.put(clazz, usedInfo);
150    
151                // Has this class be bind?
152                ret = findClassBindings(clazz, knownInstances);
153                if (ret != null) {
154                    return ret;
155                }
156    
157                // Do we need to create a string?
158                if (clazz == String.class) {
159                    return (T) RandomCreator.getRandomString();
160                }
161    
162                // Is the class an enum?
163                if (clazz.isEnum()) {
164                    T[] enums = clazz.getEnumConstants();
165                    return enums[RandomCreator.getRandomInt(enums.length - 1)];
166                }
167                // Load the constructors
168                List<Constructor<?>> consts = constructorCache.getCachedConstructors(clazz);
169                if (consts == null) {
170                    consts = new ArrayList<Constructor<?>>();
171                    Constructor<?>[] _con = clazz.getConstructors();
172                    // Sort the constructors by their parameter-count
173                    java.util.Arrays.sort(_con, new ConstructorComparator());
174                    consts.addAll(Arrays.asList(_con));
175                    // Add to cache
176                    constructorCache.addConstructors(clazz, consts);
177                }
178                // Check if we have a prefered Constructor and try it
179                Constructor<T> preferedConstructor = (Constructor<T>) constructorCache.getPreferedConstructor(clazz);
180                if (preferedConstructor != null) {
181                    ret = tryConstructor(preferedConstructor, knownInstances);
182                }
183                if (ret == null) {
184                    for (Constructor<?> co : consts) {
185                        ret = (T) tryConstructor(co, knownInstances);
186                        if (ret != null) {
187                            constructorCache.setPreferedConstructor(clazz, co);
188                            // Worked
189                            break;
190                        }
191    
192                    }
193                }
194                if (ret == null) {
195                    throw new IllegalArgumentException("The given class couldn't be instantiated");
196                }
197    
198                usedInfo.setInstance(ret);
199                usedInfo.setPopulated(true);
200                populateObject(ret, knownInstances);
201    
202                return ret;
203            } else {
204                return (T) knownInstances.get(clazz).getInstance();
205            }
206        }
207    
208        /**
209         * Tries to instantiate an object with the given constructor.
210         */
211        private <T> T tryConstructor(final Constructor<T> c, final Map<Class<?>, ClassUsageInfo<?>> knownInstances) {
212            @SuppressWarnings("unchecked")
213            Class<T>[] parameters = (Class<T>[]) c.getParameterTypes();
214            try {
215                if (parameters.length > 0) {
216                    final Object[] params = new Object[parameters.length];
217                    for (int i = 0; i < params.length; i++) {
218                        params[i] = create(parameters[i], knownInstances);
219                    }
220                    return c.newInstance(params);
221                } else {
222                    return c.newInstance();
223                }
224            } catch (InvocationTargetException e) {
225                logger.trace(String.format("unable to invoke constructor '%s', string should represent a number?", c.getName()));
226            } catch (InstantiationException e) {
227                logger.debug(String.format("unable to invoke constructor '%s'", c.getName()), e);
228            } catch (IllegalAccessException e) {
229                logger.debug(String.format("unable to invoke constructor '%s'", c.getName()), e);
230            }
231            return null;
232        }

234        /**
235         * Populates given object with dummy value. The behavior can vary depending on the type of object, as follows:
236         * <ul>
237         * <li><strong>Collection</strong>: if the type is a subtype of {@link Collection}, a random number of items, 1 or 2, will be added</li>
238         * <li><strong>Map</strong>: if the type is a subtype of {@link Map}, a random number of key/value entries, 1 or 2, will be added</li>
239         * <li><strong>Other types</strong>: The <em>setter</em> methods will be retrieved from the object and will be invoked with a new dummy
240         * value.</li>
241         * </ul>
242         * 
243         * @param subject The object to populate with dummy values.
244         * @param knownInstances A list of known instances to keep track of already processed classes (to avoid infinite loop)
245         * @see #create(Class, Map)
246         */
247        @SuppressWarnings({ "unchecked" })
248        private <T> void populateObject(final T subject, final Map<Class<?>, ClassUsageInfo<?>> knownInstances) {
249            final Class<?> clazz = subject.getClass();
250    
251            if (subject instanceof Collection) {
252                for (int i = 0; i < RandomCreator.getRandomInt(2) + 1; i++) {
253                    // detect generic declarations
254                    Type[] genericTypes = ((ParameterizedType) subject.getClass().getGenericSuperclass()).getActualTypeArguments();
255                    if (genericTypes.length > 0 && (genericTypes[0] instanceof Class)) {
256                        // uses generic type if available, Integer for '<T extends List<Integer>>'
257                        ((Collection<Object>) subject).add(create((Class<?>) genericTypes[0], knownInstances));
258                    } else {
259                        // use default String value for raw type Collection
260                        ((Collection<Object>) subject).add(create(String.class, knownInstances));
261                    }
262                }
263            } else if (subject instanceof Map) {
264                for (int i = 0; i < RandomCreator.getRandomInt(2) + 1; i++) {
265                    // detect generic declarations
266                    Type[] genericTypes = ((ParameterizedType) subject.getClass().getGenericSuperclass()).getActualTypeArguments();
267                    if (genericTypes.length > 0 && (genericTypes[0] instanceof Class)) {
268                        // uses generic type if available, String and Integer for '<T extends Map<String, Double>>'
269                        ((Map<Object, Object>) subject).put(create((Class<?>) genericTypes[0], knownInstances), create((Class<?>) genericTypes[1], knownInstances));
270                    } else {
271                        // use default String and String value for raw type Collection
272                        ((Map<Object, Object>) subject).put(create(String.class, knownInstances), create(String.class, knownInstances));
273                    }
274                }
275            } else {
276                List<Method> setter = discoverSetters(clazz);
277    
278                Object parameter = null;
279                for (Method m : setter) {
280                    // Load the parameter to pass to this method
281                    parameter = create(m.getParameterTypes()[0], knownInstances);
282                    try {
283                        m.invoke(subject, parameter);
284                    } catch (Exception e) {
285                        e.printStackTrace();
286                    }
287                }
288            }
289        }
290    
291        private List<Method> discoverSetters(final Class<?> clazz) {
292            List<Method> setter = methodCache.getSetterForClass(clazz);
293            if (setter == null) {
294                setter = new ArrayList<Method>();
295                Map<Class<?>, List<FieldWrapper>> fields = FieldUtils.collectFields(clazz, Object.class, EnumSet.allOf(Visibility.class), EnumSet.of(BeanRestriction.YES_SETTER));
296                for (List<FieldWrapper> fieldWrappers : fields.values()) {
297                    for (FieldWrapper fieldWrapper : fieldWrappers) {
298                        setter.add(fieldWrapper.getSetter());
299                    }
300                }
301                methodCache.addSetterForClass(clazz, setter);
302            }
303            return setter;
304        }

306        @SuppressWarnings("unchecked")
307        private <T> T findClassBindings(final Class<T> clazz, final Map<Class<?>, ClassUsageInfo<?>> used_classes) {
308            Object bind = classBindings.find(clazz);
309            if (bind != null) {
310                // Do we have a constructor binding?
311                if (bind instanceof Constructor) {
312                    return tryConstructor((Constructor<T>) bind, used_classes);
313                } else if (bind instanceof Method) {
314                    Method m = (Method) bind;
315                    Class<?>[] parameters = m.getParameterTypes();
316                    final Object[] params = new Object[parameters.length];
317                    for (int i = 0; i < params.length; i++) {
318                        params[i] = create(parameters[i], used_classes);
319                    }
320                    try {
321                        return (T) m.invoke(null, params);
322                    } catch (InvocationTargetException e) {
323                        logger.debug(String.format("failed to invoke Method [%s] to product an object of type [%s]", m.getName(), clazz), e);
324                    } catch (IllegalAccessException e) {
325                        logger.debug(String.format("failed to invoke Method [%s] to product an object of type [%s]", m.getName(), clazz), e);
326                    }
327                    return null;
328                } else if (bind.getClass() == Class.class) {
329                    return create((Class<? extends T>) bind, used_classes);
330                } else {
331                    return (T) bind;
332                }
333            } else {
334                return null;
335            }
336        }

338        /**
339         * This method checks, if the class, we want to create is a special one (primitives and String)
340         * 
341         * @param <T>
342         * @param clazz
343         * @return
344         */
345        @SuppressWarnings("unchecked")
346        private <T> T checkPrimitivesAndArray(final Class<T> clazz) {
347            // Check if we have a primitive or string
348            if (clazz.isPrimitive()) {
349                return buildPrimitive(clazz);
350            }
351            // Do we have an array?
352            if (clazz.isArray()) {
353                int length = RandomCreator.getRandomInt(2) + 1;
354                Object parameter = Array.newInstance(clazz.getComponentType(), length);
355                for (int i = 0; i < length; i++) {
356                    Array.set(parameter, i, create(clazz.getComponentType()));
357                }
358                return (T) parameter;
359            }
360            return null;
361        }

363        /**
364         * Method to build a primitive dummy.
365         * 
366         * @see RandomCreator
367         * 
368         * @param c The primitive type to be created.
369         * @return A primitive dummy value.
370         */
371        @SuppressWarnings("unchecked")
372        private static <T> T buildPrimitive(Class<T> c) {
373            if (c == (java.lang.Integer.TYPE)) {
374                return (T) (Integer) RandomCreator.getRandomInt();
375            } else if (c == (java.lang.Long.TYPE)) {
376                return (T) (Long) RandomCreator.getRandomLong();
377            } else if (c == (java.lang.Float.TYPE)) {
378                return (T) (Float) RandomCreator.getRandomFloat();
379            } else if (c == (java.lang.Boolean.TYPE)) {
380                return (T) (Boolean) RandomCreator.getRandomBoolean();
381            } else if (c == (java.lang.Character.TYPE)) {
382                return (T) (Character) RandomCreator.getRandomChar();
383            } else if (c == (java.lang.Byte.TYPE)) {
384                return (T) (Byte) RandomCreator.getRandomByte();
385            } else if (c == (java.lang.Short.TYPE)) {
386                return (T) (Short) RandomCreator.getRandomShort();
387            } else if (c == (java.lang.Double.TYPE)) {
388                return (T) (Double) RandomCreator.getRandomDouble();
389            }
390            return null;
391        }

393        /**
394         * Comparator to sort constructors by their number of parameters
395         */
396        private static class ConstructorComparator implements Comparator<Constructor<?>> {
397    
398            public int compare(Constructor<?> o1, Constructor<?> o2) {
399                int num_o1 = o1.getParameterTypes().length;
400                int num_o2 = o2.getParameterTypes().length;
401                if (num_o1 < num_o2) {
402                    return -1;
403                }
404                if (num_o1 == num_o2) {
405                    return 0;
406                }
407                return 1;
408            }
409        }
