018    package org.dummycreator;
019    
020    import java.util.Random;
021    
022    import de.svenjacobs.loremipsum.LoremIpsum;
023    
024    /**
025     * 
026     * @author Alexander Muthmann <amuthmann@dev-eth0.de>
027     * @version 04/2010
028     */
029    class RandomCreator {
030    
031        private static final Random r = new Random();
032        private static final LoremIpsum loremIpsum = new LoremIpsum();
033    
034        public static String getRandomString() {
035            String gibberish = "";
036            for (int i = 0; i <= r.nextInt(3); i++) {
037                gibberish += (gibberish.length() != 0 ? " " : "") + loremIpsum.getWords(1, r.nextInt(50));
038            }
039            return gibberish.replaceAll("\\s", "_").replaceAll("[\\.\\,]", "");
040        }
041    
042        public static boolean getRandomBoolean() {
043            return r.nextBoolean();
044        }
045    
046        public static int getRandomInt() {
047            return r.nextInt();
048        }
049    
050        public static char getRandomChar() {
051            return (char) r.nextInt();
052        }
053    
054        public static byte getRandomByte() {
055            return (byte) r.nextInt();
056        }
057    
058        public static long getRandomLong() {
059            return r.nextLong();
060        }
061    
062        public static float getRandomFloat() {
063            return r.nextFloat();
064        }
065    
066        public static double getRandomDouble() {
067            return r.nextDouble();
068        }
069    
070        public static short getRandomShort() {
071            return (short) r.nextInt();
072        }
073    
074        public static int getRandomInt(int max) {
075            return r.nextInt(max);
076        }
077    }
